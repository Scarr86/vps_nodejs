Привет Express!

  1. create rep github
  2. install git to server
  3. git clone vps_server
  4. install node to server
  5. install npm to server
  6. install nginx to server
  7. backup /etc/nginx/sites-available/default to /etc/nginx/sites-available/bk.default
  8. edit /etc/nginx/sites-available/default add: 

                        location / {
                        proxy_pass http://localhost:3000;
                        proxy_http_version 1.1;
                        proxy_set_header Upgrade $http_upgrade;
                        proxy_set_header Connection 'upgrade';
                        proxy_set_header Host $host;
                        proxy_cache_bypass $http_upgrade;
                    }

  9. cd ~/opt/vps_nodejs/
  10. test: node server.js
  11. sudo npm install pm2 -g (auto start node server)
  12. sudo pm2 start server.js
